\begin{mystatement}{proposition}{normal::uniqueness}
    Let $N \sub M$ be an embedded submanifold.
    Suppose there exists a normal microbundle $\bn$ of $N$ in $M$.
    Then $\bt_N \oplus \bn \cong \bt_M\restr{N}$.
\end{mystatement}
