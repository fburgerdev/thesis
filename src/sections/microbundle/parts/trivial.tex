\begin{mystatement}{example}[trivial microbundle]{microbundle::trivial}[55]
    Given a topological space $B$, the \defterm{standard microbundle} $\be{B}$ over $B$ is a microbundle
    \[ \bundle{B}{B \cross \R^n}{\iota}{\pi_1} \]
    where $\iota(b) = (b, 0)$ and $\pi_1(b, x) = b$.
    A microbundle $\bb$ over $B$ is \defterm{trivial} if it is isomorphic to $\be{B}$.
\end{mystatement}
