% definition rooted
\begin{mydefinition}{suspension::rooted}
    % rooted
    A \defterm{rooted microbundle} $\bb$ over $B$ is a microbundle over $B$ together with an isomorphism-germ
    \[ R: \bbb \double \be_{b_0}. \]
    % ismorphy
    Two rooted microbundles $\bb$ and $\bb'$ are \defterm{isomorphic} if there exists an isomorphism germ $\bb \double \bb'$ extending
    \[ \germdef{\inv{R'} \circ R}{\bbb}{\bbb'}. \]
\end{mydefinition}
