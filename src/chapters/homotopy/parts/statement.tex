% theorem
\begin{mytheorem}[Homotopy Theorem]{homotopy::theorem}
    Let $\bb$ be a microbundle over $B$ and let $f, g: A \to B$ be two maps where $A$ is paracompact.
    If $f$ and $g$ are homotopic, then $\ind{f}\bb$ and $\ind{g}\bb$ are isomorphic.
\end{mytheorem}
