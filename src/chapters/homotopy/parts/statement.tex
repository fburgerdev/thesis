% homotopy theorem
\begin{mytheorem}[Homotopy Theorem]{homotopy::theorem}%
    Let $\bb$ be a microbundle over $B$ and $f, g: A \to B$ be two maps.
    If $f$ and $g$ are homotopic, then $f^*\bb$ and $g^*\bb$ are isomorphic.
\end{mytheorem}
