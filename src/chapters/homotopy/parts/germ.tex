% map-germ
\begin{mydefinition}[map-germ]
    A \defterm{map-germ} $F: (X, A) \Rightarrow (Y, B)$ between topological pairs $(X, A)$ and $(Y, B)$ is an equivalence class of maps $f: (X, A) \to (Y, B)$
    where $f \sim g :\iff f\restr{U} = g\restr{U}$ for some neighborhood $U \sub X$ of $A$. 
\end{mydefinition}

% homeomorphism-germ
\begin{myparagraph}
    A \defterm{homeomorphism-germ} $F: (X, A) \Rightarrow (Y, B)$ is a map-germ
    such that a representative $f: U_f \to Y$ maps homeomorphically on a neighborhood of $B$.  
\end{myparagraph}
