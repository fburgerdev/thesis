% map-germ
\begin{mydefinition}[map-germ]{homotopy::germ}
    A \defterm{map-germ} $F: (X, A) \double (Y, B)$ between topological pairs $(X, A)$ and $(Y, B)$ is an equivalence class of maps $(X, A) \to (Y, B)$
    where $f \sim g :\iff f\restr{U} = g\restr{U}$ for an arbitrary neighborhood $U \sub X$ of $A$. 
\end{mydefinition}

% homeomorphism-germ
\begin{myparagraph}
    A \defterm{homeomorphism-germ} $F: (X, A) \double (Y, B)$ is a map-germ
    such that there exists a representative $f: U_f \to Y$ that maps homeomorphically to a neighborhood of $B$.  
\end{myparagraph}
